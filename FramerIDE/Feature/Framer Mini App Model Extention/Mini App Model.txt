**2. The Logic (`src/extension.ts`)**
Replace the entire file with this code. This creates the Sidebar Provider.

```typescript
import * as vscode from 'vscode';

export function activate(context: vscode.ExtensionContext) {
	const provider = new BaseFrameProvider(context.extensionUri);

	context.subscriptions.push(
		vscode.window.registerWebviewViewProvider(BaseFrameProvider.viewType, provider)
	);

	context.subscriptions.push(
		vscode.commands.registerCommand('base-frame-preview.refresh', () => {
			provider.refresh();
		})
	);
}

class BaseFrameProvider implements vscode.WebviewViewProvider {

	public static readonly viewType = 'base-frame-view';
	private _view?: vscode.WebviewView;

	constructor(
		private readonly _extensionUri: vscode.Uri,
	) { }

	public resolveWebviewView(
		webviewView: vscode.WebviewView,
		context: vscode.WebviewViewResolveContext,
		_token: vscode.CancellationToken,
	) {
		this._view = webviewView;

		webviewView.webview.options = {
			enableScripts: true,
			localResourceRoots: [
				this._extensionUri
			]
		};

		webviewView.webview.html = this._getHtmlForWebview(webviewView.webview);
	}

	public refresh() {
		if (this._view) {
			this._view.webview.html = this._getHtmlForWebview(this._view.webview);
		}
	}

	private _getHtmlForWebview(webview: vscode.Webview) {
		// This is the UI of your Sidebar
		return `<!DOCTYPE html>
		<html lang="en">
		<head>
			<meta charset="UTF-8">
			<meta name="viewport" content="width=device-width, initial-scale=1.0">
			<style>
				body { padding: 0; margin: 0; background-color: #1e1e1e; display: flex; flex-direction: column; height: 100vh; align-items: center; }
				
				/* The "Phone" Container */
				.phone-frame {
					width: 320px;
					height: 600px;
					background: #fff;
					border-radius: 30px;
					border: 8px solid #333;
					overflow: hidden;
					margin-top: 20px;
					position: relative;
					box-shadow: 0 10px 25px rgba(0,0,0,0.5);
				}

				/* Farcaster Header Mockup */
				.fc-header {
					height: 50px;
					background: #f3f3f3;
					display: flex;
					align-items: center;
					padding: 0 15px;
					border-bottom: 1px solid #ddd;
					font-family: -apple-system, BlinkMacSystemFont, 'Segoe UI', Roboto, sans-serif;
					font-weight: bold;
					color: #444;
				}

				iframe {
					width: 100%;
					height: calc(100% - 50px);
					border: none;
				}

				.controls {
					width: 100%;
					padding: 10px;
					display: flex;
					gap: 5px;
					justify-content: center;
				}

				input {
					width: 70%;
					padding: 5px;
					border-radius: 4px;
					border: 1px solid #444;
					background: #252526;
					color: white;
				}

				button {
					background: #0052FF; /* Base Blue */
					color: white;
					border: none;
					padding: 5px 10px;
					border-radius: 4px;
					cursor: pointer;
				}
			</style>
		</head>
		<body>
			<div class="controls">
				<input type="text" id="urlInput" value="http://localhost:3000" />
				<button onclick="loadUrl()">Load</button>
			</div>

			<div class="phone-frame">
				<div class="fc-header">
					üîµ Base Frame
				</div>
				<iframe id="frame" src="http://localhost:3000"></iframe>
			</div>

			<script>
				const vscode = acquireVsCodeApi();
				
				function loadUrl() {
					const url = document.getElementById('urlInput').value;
					const frame = document.getElementById('frame');
					frame.src = url;
				}
			</script>
		</body>
		</html>`;
	}
}
```

#### **Step 3: Test It (Run the Demo)**

1.  Press **F5** in VS Code.
2.  A new VS Code window will open (this is the "Extension Development Host").
3.  In that new window, look at the **Activity Bar** (the far left icons). You should see a new blank icon (or the Base logo if you added it).
4.  Click it. You will see "Frame Simulator".
5.  Start a local Next.js project in your terminal (`npm run dev` -> localhost:3000).
6.  Click "Load" in your extension.
7.  **BOOM.** Your Base App is running inside the VS Code sidebar in a mobile skin.
